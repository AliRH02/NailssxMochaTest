 <script>
const images = document.querySelectorAll(".gallery img");
let currentIndex = 0;

let startX = 0;
let startY = 0;

let isSwiping = false;

const lightbox = document.getElementById("lightbox");
const track = document.getElementById("lightbox-track");
const dotsContainer = document.getElementById("dots");

let currentTranslate = 0;
let prevTranslate = 0;

function openImage(img) {
  currentIndex = Array.from(images).indexOf(img);

  currentTranslate = 0;
  prevTranslate = 0;

  track.innerHTML = "";
  dotsContainer.innerHTML = "";

  images.forEach((image, i) => {
    const clone = image.cloneNode();
    track.appendChild(clone);

    const dot = document.createElement("span");
    if (i === currentIndex) dot.classList.add("active");
    dotsContainer.appendChild(dot);
  });

  lightbox.style.display = "flex";

  setTimeout(() => {
    lightbox.classList.add("active");
    updateSlide();

  }, 10);
}

function updateSlide() {
 console.log("Current Index:", currentIndex);
  console.log("Is First Slide?", currentIndex === 0);
  console.log("Is Last Slide?", currentIndex === images.length - 1);
	
if(currentIndex<0 || currentIndex > images.length-1){
console.log("Slide update blocked. Index out of bounds:", currentIndex);
    return;   
}
const slideWidth = lightbox.offsetWidth;


currentTranslate = -currentIndex * slideWidth;
  prevTranslate = currentTranslate;

  track.style.transition = "transform 0.4s ease";
  track.style.transform = `translateX(${currentTranslate}px)`;

  const dots = document.querySelectorAll(".dots span");
  dots.forEach((d, i) => {
    d.classList.toggle("active", i === currentIndex);
  
  
}
);
}

function showNext(e) {
  if (e) e.stopPropagation();
  currentIndex = (currentIndex + 1) % images.length;
  updateSlide();
}

function showPrev(e) {
  if (e) e.stopPropagation();
  currentIndex = (currentIndex - 1 + images.length) % images.length;
  updateSlide();
}

function closeImage() {
  const lightbox = document.getElementById("lightbox");
  lightbox.classList.remove("active");

  setTimeout(() => {
    lightbox.style.display = "none";
  }, 300);
}


lightbox.addEventListener("click", function(e) {
  // only close if clicking the background itself
  if (e.target === lightbox) {
    closeImage();
  }
});


lightbox.addEventListener("touchstart", (e) => {
  if (e.target.tagName !== "IMG") return;

  isSwiping = false;

  startX = e.touches[0].clientX;
  startY = e.touches[0].clientY;
});


lightbox.addEventListener("touchmove", (e) => {
  if (e.target.tagName !== "IMG") return;

  const moveX = e.touches[0].clientX;
  const diff = moveX - startX;

  // detect real swipe
  if (Math.abs(diff) > 10) {
    isSwiping = true;
  }

  const slideWidth = lightbox.offsetWidth;

  const maxTranslate = 0;
  const minTranslate = -slideWidth * (images.length - 1);

  currentTranslate = prevTranslate + diff;

  // clamp (prevents empty space scrolling)
  currentTranslate = Math.max(
    minTranslate,
    Math.min(maxTranslate, currentTranslate)
  );

  track.style.transition = "none";
  track.style.transform = `translateX(${currentTranslate}px)`;
});

lightbox.addEventListener("touchend", () => {
  const movedBy = currentTranslate - prevTranslate;

  if (movedBy < -60) currentIndex++;
  if (movedBy > 60) currentIndex--;

  currentIndex = Math.max(0, Math.min(currentIndex, images.length - 1));

  const slideWidth = lightbox.offsetWidth;
  prevTranslate = -currentIndex * slideWidth;

  track.style.transition = "transform 0.4s ease";
  updateSlide();
isSwiping = false;
});


function getDistance(touches) {
  return Math.hypot(
    touches[0].clientX - touches[1].clientX,
    touches[0].clientY - touches[1].clientY
  );
}


document.addEventListener("keydown", (e) => {
  if (e.key === "ArrowRight") showNext();
  if (e.key === "ArrowLeft") showPrev();
  if (e.key === "Escape") closeImage();
});
</script>
